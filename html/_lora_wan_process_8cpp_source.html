47 template <template <class R> class T, class RADIOTYPE>
48 eLoraWan_Process_States LoraWanObject <T,RADIOTYPE> ::LoraWanProcess( uint8_t* AvailableRxPacket ) {
49 
50  *AvailableRxPacket = NO_LORA_RXPACKET_AVAILABLE;
51  #if LOW_POWER_MODE == 1
52  if ( ( IsJoined ( ) == NOT_JOINED ) && ( mcu.RtcGetTimeSecond( ) < packet.RtcNextTimeJoinSecond ) ){
53  DEBUG_PRINTF("TOO SOON TO JOIN time is %d time target is : %d \n",mcu.RtcGetTimeSecond( ), packet.RtcNextTimeJoinSecond);
54  StateLoraWanProcess = LWPSTATE_IDLE ;
55  }
56  #endif
57  if ( ( mcu.RtcGetTimeSecond( ) - FailSafeTimestamp ) > 120 ) {
58  RadioReset ( ) ;
59  StateLoraWanProcess = LWPSTATE_ERROR ;
60  DEBUG_MSG ( "ERROR : FAILSAFE EVENT OCCUR \n");
61  // NVIC_SystemReset() move into the user main;
62  }
63  switch ( StateLoraWanProcess ) {
64  /************************************************************************************/
65  /* STATE IDLE */
66  /************************************************************************************/
67  case LWPSTATE_IDLE :
68  break;
69  /************************************************************************************/
70  /* STATE TX */
71  /************************************************************************************/
72  case LWPSTATE_SEND:
73  switch ( GetRadioState( ) ) {
74 
75  case RADIOSTATE_IDLE :
76  AttachRadioIsr ( );
77  packet.ConfigureRadioAndSend( );
78  DEBUG_MSG( "\n" );
79  DEBUG_MSG( " **************************\n " );
80  DEBUG_MSG( " * Send Payload *\n " );
81  DEBUG_MSG( " **************************\n " );
82  break ;
83 
84  case RADIOSTATE_TXFINISHED :
85  packet.ConfigureTimerForRx ( RX1 );
86  StateLoraWanProcess = LWPSTATE_RX1;
87  break ;
88 
89  default :
90  break;
91  }
92  break;
93  /************************************************************************************/
94  /* STATE RX1 */
95  /* RX1DELAY is defined in ms */
96  /************************************************************************************/
97  case LWPSTATE_RX1:
98  if ( GetRadioState( ) == RADIOSTATE_RX1FINISHED ) {
99  if ( GetRadioIrqFlag ( ) == RECEIVE_PACKET_IRQ_FLAG) {
100  DEBUG_MSG( "\n" );
101  DEBUG_MSG( " **************************\n " );
102  DEBUG_MSG( " * Receive a downlink RX1 *\n " );
103  DEBUG_MSG( " **************************\n " );
104  StateLoraWanProcess = LWPSTATE_PROCESSDOWNLINK;
105  } else {
106  DEBUG_MSG( "\n" );
107  DEBUG_MSG( " **************************\n " );
108  DEBUG_MSG( " * RX1 Timeout *\n " );
109  DEBUG_MSG( " **************************\n " );
110  packet.ConfigureTimerForRx ( RX2 );
111  StateLoraWanProcess = LWPSTATE_RX2;
112  }
113  }
114  break;
115  /************************************************************************************/
116  /* STATE RX2 */
117  /************************************************************************************/
118  case LWPSTATE_RX2:
119 
120  if ( GetRadioState( ) == RADIOSTATE_IDLE ) {
121  if ( GetRadioIrqFlag ( ) == RECEIVE_PACKET_IRQ_FLAG) {
122  DEBUG_MSG( "\n" );
123  DEBUG_MSG( " **************************\n " );
124  DEBUG_MSG( " * Receive a downlink RX2 *\n " );
125  DEBUG_MSG( " **************************\n " );
126  StateLoraWanProcess = LWPSTATE_PROCESSDOWNLINK;
127  } else {
128  DEBUG_MSG( "\n" );
129  DEBUG_MSG( " **************************\n " );
130  DEBUG_MSG( " * RX2 Timeout *\n " );
131  DEBUG_MSG( " **************************\n " );
132  StateLoraWanProcess = LWPSTATE_UPDATEMAC;
133  }
134  }
135  break;
136  /************************************************************************************/
137  /* STATE PROCESS DOWNLINK */
138  /* At this step crc is valid */
139  /* Step 1 : CheckRxPayloadLength */
140  /* Step 2 : ExtractRxMhdr */
141  /* Step 3 : ExtractRxFhdr */
142  /* Step 4 : Check Mic */
143  /* Step 5 : Decrypt Payload */
144  /* Step 6 : Extract Fport to select Between App/nwm Payload */
145  /************************************************************************************/
146  case LWPSTATE_PROCESSDOWNLINK:
147  DEBUG_MSG( "\n" );
148  DEBUG_MSG( " **************************\n " );
149  DEBUG_MSG( " * Process Downlink *\n " );
150  DEBUG_MSG( " **************************\n " );
151  ValidRxPacket = packet.DecodeRxFrame( ); // return NOVALIDRXPACKET or USERRX_FOPTSPACKET or NWKRXPACKET or JOIN_ACCEPT_PACKET.
152  StateLoraWanProcess = LWPSTATE_UPDATEMAC;
153  break;
154  /************************************************************************************/
155  /* STATE UPDATE MAC */
156  /************************************************************************************/
157  case LWPSTATE_UPDATEMAC:
158  DetachRadioIsr ( );
159  packet.Phy.StateRadioProcess = RADIOSTATE_IDLE;
160  DEBUG_MSG( "\n" );
161  DEBUG_MSG( " **************************\n " );
162  DEBUG_MSG( " * UpdateMac *\n " );
163  DEBUG_MSG( " **************************\n " );
164  if ( ValidRxPacket == JOIN_ACCEPT_PACKET){
165  packet.UpdateJoinProcedure( );
166  packet.RegionSetDataRateDistribution( packet.AdrModeSelect);//@note because datarate Distribution has been changed during join
167  }
168  if ( ( ValidRxPacket == NWKRXPACKET) || ( ValidRxPacket == USERRX_FOPTSPACKET) ) {
169  packet.ParseManagementPacket( );
170  }
171  packet.UpdateMacLayer();
172  *AvailableRxPacket = packet.AvailableRxPacketForUser;
173  if ( ( packet.IsFrameToSend == NWKFRAME_TOSEND ) || ( packet.IsFrameToSend == USRFRAME_TORETRANSMIT) ) {// @note ack send during the next tx|| ( packet.IsFrameToSend == USERACK_TOSEND ) ) {
174  packet.IsFrameToSend = NOFRAME_TOSEND;
175  RtcTargetTimer = mcu.RtcGetTimeSecond( ) + randr( 2, 6 );
176  StateLoraWanProcess = LWPSTATE_TXwait;
177  } else {
178  RadioReset ( ) ;
179  StateLoraWanProcess = LWPSTATE_IDLE;
180  }
181  ValidRxPacket = NO_MORE_VALID_RX_PACKET;
182  break;
183  /************************************************************************************/
184  /* STATE TXwait MAC */
185  /************************************************************************************/
186  case LWPSTATE_TXwait:
187  DEBUG_MSG(".");
188  if ( mcu.RtcGetTimeSecond( ) > RtcTargetTimer) {
189  StateLoraWanProcess = LWPSTATE_SEND; //@note the frame have already been prepare in Upadate Mac Layer
190  }
191  break;
192 
193  default:
194  DEBUG_MSG( " Illegal state\n " );
195  break;
196  }
197  return ( StateLoraWanProcess );
198 }

252 eLoraWan_Process_States LoraWanObject <T,RADIOTYPE> ::SendPayload ( uint8_t fPort, const uint8_t* dataIn, const uint8_t sizeIn, uint8_t PacketType ) {
253  eStatusLoRaWan status;
254  FailSafeTimestamp = mcu.RtcGetTimeSecond( ) ;
255  packet.RegionGiveNextDataRate ( ); // both choose the next tx data rate but also compute the Sf and Bw (region )
256  status = packet.RegionMaxPayloadSize ( sizeIn );
257  if ( status == ERRORLORAWAN ) {
258  DEBUG_MSG( " ERROR : PAYLOAD SIZE TOO HIGH \n" );
259  return ( LWPSTATE_ERROR );
260  }
261  if ( GetIsOtaDevice ( ) == OTA_DEVICE ) {
262  if ( packet.Phy.JoinedStatus == NOT_JOINED ) {
263  DEBUG_MSG( " ERROR : OTA DEVICE NOT JOINED YET\n" );
264  return ( LWPSTATE_ERROR );
265  }
266  }
267  RadioReset ( ) ;
268  CopyUserPayload( dataIn,sizeIn );
269  packet.UserPayloadSize = sizeIn;
270  packet.fPort = fPort;
271  packet.MType = PacketType;
272 
273  packet.BuildTxLoraFrame( );
274  packet.EncryptTxFrame( );
275  if (PacketType == CONF_DATA_UP){
276  packet.MacNbTransCpt = MAX_CONFUP_MSG;
277  } else {
278  packet.MacNbTransCpt = packet.MacNbTrans;
279  }
280  StateLoraWanProcess = LWPSTATE_SEND;
281  return( StateLoraWanProcess );

55 IrqFlags_t SX126x::GetIrqFlags( void ) {
56  uint8_t irqStatus[2];
57  IrqFlags_t irqFlags = RADIO_IRQ_NONE;
58 
59  // Read IRQ status
60  ReadCommand( GET_IRQ_STATUS, irqStatus, 2 );
61  // Parse it
62  if ( ( irqStatus[0] & ( IRQ_RX_TX_TIMEOUT >> 8 ) ) !=0 ) {
63  irqFlags = (IrqFlags_t) (irqFlags | RXTIMEOUT_IRQ_FLAG);
64  }
65  if ( ( irqStatus[1] & IRQ_RX_DONE ) !=0 ) {
66  irqFlags = (IrqFlags_t) (irqFlags | RECEIVE_PACKET_IRQ_FLAG);
67  }
68  /* Not used by the MAC for now
69  if ( ( irqStatus[1] & IRQ_TX_DONE ) !=0 ) {
70  irqFlags = (IrqFlags_t) (irqFlags | RADIO_IRQ_TX_DONE);
71  }
72  */
73  if ( ( ( irqStatus[1] & IRQ_HEADER_ERROR ) != 0 ) ||
74  ( ( irqStatus[1] & IRQ_CRC_ERROR ) != 0 ) ) {
75  irqFlags = (IrqFlags_t) (irqFlags | BAD_PACKET_IRQ_FLAG);
76  }
77 
78  return irqFlags;
79 }

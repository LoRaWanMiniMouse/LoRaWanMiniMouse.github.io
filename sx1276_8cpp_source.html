118 void SX1276::CalibrateImage( void )
119 {
120  uint8_t regPaConfigInitVal;
121  uint32_t initialFreq;
122 
123  // Save context
124  regPaConfigInitVal = Read( REG_PACONFIG );
125  initialFreq = ( double )( ( ( uint32_t )this->Read( REG_FRFMSB ) << 16 ) |
126  ( ( uint32_t )this->Read( REG_FRFMID ) << 8 ) |
127  ( ( uint32_t )this->Read( REG_FRFLSB ) ) ) * ( double )FREQ_STEP;
128 
129  // Cut the PA just in case, RFO output, power = -1 dBm
130  this->Write( REG_PACONFIG, 0x00 );
131 
132  // Launch Rx chain calibration for LF band
133  Write ( REG_IMAGECAL, ( Read( REG_IMAGECAL ) & RF_IMAGECAL_IMAGECAL_MASK ) | RF_IMAGECAL_IMAGECAL_START );
134  while( ( Read( REG_IMAGECAL ) & RF_IMAGECAL_IMAGECAL_RUNNING ) == RF_IMAGECAL_IMAGECAL_RUNNING )
135  {
136  }
137 
138  // Sets a Frequency in HF band
139  SetRfFrequency( 868000000 );
140 
141  // Launch Rx chain calibration for HF band
142  Write ( REG_IMAGECAL, ( Read( REG_IMAGECAL ) & RF_IMAGECAL_IMAGECAL_MASK ) | RF_IMAGECAL_IMAGECAL_START );
143  while( ( Read( REG_IMAGECAL ) & RF_IMAGECAL_IMAGECAL_RUNNING ) == RF_IMAGECAL_IMAGECAL_RUNNING )
144  {
145  }
146 
147  // Restore context
148  this->Write( REG_PACONFIG, regPaConfigInitVal );
149  SetRfFrequency( initialFreq );
150 }

166 void SX1276::SetModulationParamsTx( uint8_t SF, eBandWidth BW, int8_t power ) {
167 
168  uint8_t LowDatarateOptimize;
169  uint8_t ValueTemp;
170  power = ( power > 20 ) ? 20 : power;
171 
172  /* Power Setting registers*/
173  if ( PA_BOOST_CONNECTED == 1 ) {
174  power = ( power > 20 ) ? 20 : power;
175  power = ( power < 2 ) ? 2 : power;
176  ValueTemp = ( power > 17 ) ? 0x87 : 0x84; // Enable/disable +20dbM option ON PA_BOOSt pin
177  Write( REG_PADAC, ValueTemp );
178  ValueTemp = ( power > 17 ) ? 0xf0 + ( power - 5 ) : 0xf0 + ( power - 2 );
179  Write( REG_PACONFIG, ValueTemp );
180  } else {
181  power = ( power > 14 ) ? 14 : power;
182  power = ( power < -1 ) ? -1 : power;
183  Write( REG_PADAC, 0x84 );
184  Write( REG_PACONFIG, ( 0x70 + power - 1 ));
185  }
186 
187  /* Set Coding rate 4/5 , Implicite Header and BW */
188  ValueTemp = 0x02 + ( ( BW + 7 ) << 4 );
189  Write( REG_LR_MODEMCONFIG1, ValueTemp );
190 
191  /* Set Enable CRC and SF */
192  ValueTemp = 4 + ( SF << 4 ) ;
193  Write( REG_LR_MODEMCONFIG2, ValueTemp ) ;
194 
195  /* Enable/disable Low datarate optimized */
196  if( ( ( BW == 0 ) && ( ( SF == 11 ) || ( SF == 12 ) ) ) || ( ( BW == 1 ) && ( SF == 12 ) ) ) {
197  LowDatarateOptimize = 0x08;
198  } else {
199  LowDatarateOptimize = 0x00;
200  }
201  Write( REG_LR_MODEMCONFIG3,LowDatarateOptimize + 4 ); // + 4 for internal AGC loop
202 
203  /* Set Preamble = 8 */
204  Write( REG_LR_PREAMBLEMSB, 0 );
205  Write( REG_LR_PREAMBLELSB, 8 );
206 
207  /* Set Normal IQ */
208  Write( REG_LR_INVERTIQ, 0x27) ;
209  Write( REG_LR_INVERTIQ2, RFLR_INVERTIQ2_OFF );
210 
211  /* Set Public sync word */
212  Write( REG_LR_SYNCWORD, 0x34);
213 }

215 void SX1276::SetModulationParamsRx( uint8_t SF, eBandWidth BW, uint16_t symbTimeout )
216 {
217  uint8_t LowDatarateOptimize;
218  uint8_t ValueTemp;
219  /* Set Coding rate 4/5 , Implicite Header and BW */
220  ValueTemp = 0x02 + ( ( BW + 7 ) << 4 );
221  Write( REG_LR_MODEMCONFIG1, ValueTemp );
222  /* Set Enable CRC and SF + MSB Timeout*/
223  ValueTemp = ( SF << 4 ) + 4 + ( ( symbTimeout >> 8 ) & 0x3 ) ;
224  Write( REG_LR_MODEMCONFIG2, ValueTemp ) ;
225  /* Set LSB Timeout*/
226  Write( REG_LR_SYMBTIMEOUTLSB, ( uint8_t )( symbTimeout & 0x00FF ) );
227 
228  /* Enable/disable Low datarate optimized */
229  if( ( ( BW == 0 ) && ( ( SF == 11 ) || ( SF == 12 ) ) ) || ( ( BW == 1 ) && ( SF == 12 ) ) ) {
230  LowDatarateOptimize = 0x08;
231  } else {
232  LowDatarateOptimize = 0x00;
233  }
234  Write( REG_LR_MODEMCONFIG3,LowDatarateOptimize + 4 ); // + 4 for internal AGC loop
235  /* Set Preamble = 8 */
236  Write( REG_LR_PREAMBLEMSB, 0 );
237  Write( REG_LR_PREAMBLELSB, 8 );
238 
239  /* Set inverted IQ */
240  Write( REG_LR_INVERTIQ, 0x67) ;
241  Write( REG_LR_INVERTIQ2, RFLR_INVERTIQ2_ON );
242 /* sensitivity optimization */
243 
244  Write( REG_LR_DETECTOPTIMIZE,( Read( REG_LR_DETECTOPTIMIZE ) & RFLR_DETECTIONOPTIMIZE_MASK ) | RFLR_DETECTIONOPTIMIZE_SF7_TO_SF12 );
245  Write( REG_LR_DETECTIONTHRESHOLD, RFLR_DETECTIONTHRESH_SF7_TO_SF12 );
246 
247  if( ( BW == 2 ) && ( Channel > RF_MID_BAND_THRESH ) )
248  {
249  // ERRATA 2.1 - Sensitivity Optimization with a 500 kHz Bandwidth
250  Write( REG_LR_TEST36, 0x02 );
251  Write( REG_LR_TEST3A, 0x64 );
252  // ERRATA 2.3 - Receiver Spurious Reception of a LoRa Signal
253  Write( REG_LR_DETECTOPTIMIZE, Read( REG_LR_DETECTOPTIMIZE ) | 0x80 );
254  }
255  else if( BW == 2 )
256  {
257  // ERRATA 2.1 - Sensitivity Optimization with a 500 kHz Bandwidth
258  Write( REG_LR_TEST36, 0x02 );
259  Write( REG_LR_TEST3A, 0x7F );
260  // ERRATA 2.3 - Receiver Spurious Reception of a LoRa Signal
261  Write( REG_LR_DETECTOPTIMIZE, Read( REG_LR_DETECTOPTIMIZE ) | 0x80 );
262  }
263  else
264  {
265  // ERRATA 2.1 - Sensitivity Optimization
266  Write( REG_LR_TEST36, 0x03 );
267  // ERRATA 2.3 - Receiver Spurious Reception of a LoRa Signal
268  Write( REG_LR_DETECTOPTIMIZE, Read( REG_LR_DETECTOPTIMIZE ) & 0x7F );
269  Write( REG_LR_TEST30, 0x00 );
270  Write( REG_LR_TEST2F, 0x40 );
271  }
272  /* Set Public sync word */
273  Write( REG_LR_SYNCWORD, 0x34);
274 }

297 void SX1276::SetRfFrequency( uint32_t frequency ) {
298  uint32_t initialFreqInt, initialFreqFrac;
299  initialFreqInt = frequency / FREQ_STEP_8;
300  initialFreqFrac = frequency - ( initialFreqInt * FREQ_STEP_8 );
301  frequency = ( initialFreqInt << 8 ) + ( ( ( initialFreqFrac << 8 ) + ( FREQ_STEP_8 / 2 ) ) / FREQ_STEP_8 );
302  Write( REG_FRFMSB, ( uint8_t )( ( frequency >> 16 ) & 0xFF ) );
303  Write( REG_FRFMID, ( uint8_t )( ( frequency >> 8 ) & 0xFF ) );
304  Write( REG_FRFLSB, ( uint8_t )( frequency & 0xFF ) );
305 }

111 template < class R >eStatusChannel LoraRegionsEU<R>::RegionBuildChannelMask ( uint8_t ChMaskCntl, uint16_t ChMask ) {
112  eStatusChannel status = OKCHANNEL;
113  switch ( ChMaskCntl ) {
114  case 0 :
115  UnwrappedChannelMask = UnwrappedChannelMask ^ ChMask;
116  for ( int i = 0 ; i < this->NUMBER_OF_CHANNEL ; i++) {
117  if ( ( ( ( UnwrappedChannelMask >> i) & 0x1 ) == 1 ) && ( this->MacTxFrequency[i] == 0) ) {
118  status = ERROR_CHANNEL_MASK ; // this status is used only for the last multiple link adr req
119  }
120  }
121  break;
122  case 6 :
123  UnwrappedChannelMask = 0;
124  for ( int i = 0 ; i < this->NUMBER_OF_CHANNEL ; i++) {
125  if ( this->MacTxFrequency[i] > 0 ) {
126  UnwrappedChannelMask = UnwrappedChannelMask ^ (1 << i ) ;
127  }
128  }
129  break;
130  default :
131  status = ERROR_CHANNEL_CNTL;
132  }
133  if ( UnwrappedChannelMask == 0 ) {
134  status = ERROR_CHANNEL_MASK ;
135  }
136  return ( status );
